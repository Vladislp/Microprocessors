LDI r16, 0xFF
OUT SPL, R16  
LDI R16, 0x10
OUT SPH, R16

LDI R16, 0xE8;LOW(1000)
LDI R17, 0x03;HIGH(1000)

CALL Delay
CALL Delay
Delay:
	PUSH R16
	PUSH R17

	ADD R16, R16
	ADC R17, R17

	STS OCR1AH, R17
	STS OCR1AL, R16

	LDI R16, 0x0D

	STS TCCR1B, R16

	LDI R16, 0x02
	Tsykkel:
		IN R17, TIFR1
		AND R17, R16
		CP R16, R17
	BRNE Tsykkel

	POP R17
	POP R16
	RET
